## Offline STT Types

### STTModelType

Supported STT model types for offline recognition.

```typescript
type STTModelType =
  | 'transducer'
  | 'nemo_transducer'
  | 'paraformer'
  | 'nemo_ctc'
  | 'wenet_ctc'
  | 'sense_voice'
  | 'zipformer_ctc'
  | 'ctc'
  | 'whisper'
  | 'funasr_nano'
  | 'fire_red_asr'
  | 'moonshine'
  | 'dolphin'
  | 'canary'
  | 'omnilingual'
  | 'medasr'
  | 'telespeech_ctc'
  | 'auto';
```

**Available at runtime:**

```typescript
import { STT_MODEL_TYPES } from 'react-native-sherpa-onnx';

console.log(STT_MODEL_TYPES);
// ['transducer', 'nemo_transducer', 'paraformer', ...]
```

### STTInitializeOptions

Options for initializing an offline STT engine.

```typescript
interface STTInitializeOptions {
  modelPath: ModelPathConfig;
  preferInt8?: boolean;
  modelType?: STTModelType;
  debug?: boolean;
  hotwordsFile?: string;
  hotwordsScore?: number;
  modelingUnit?: 'cjkchar' | 'bpe' | 'cjkchar+bpe';
  bpeVocab?: string;
  numThreads?: number;
  provider?: string;
  ruleFsts?: string;
  ruleFars?: string;
  dither?: number;
  modelOptions?: SttModelOptions;
}
```

<ParamField path="modelPath" type="ModelPathConfig" required>
  Model directory path configuration.
</ParamField>

<ParamField path="preferInt8" type="boolean">
  Prefer int8 quantized models (smaller, faster) over regular models (higher accuracy).
</ParamField>

<ParamField path="modelType" type="STTModelType" default="auto">
  Explicit model type or auto-detection.
</ParamField>

<ParamField path="debug" type="boolean" default="false">
  Enable debug logging.
</ParamField>

<ParamField path="hotwordsFile" type="string">
  Path to hotwords file (transducer/nemo_transducer only).
</ParamField>

<ParamField path="hotwordsScore" type="number" default="1.5">
  Hotwords score for keyword boosting.
</ParamField>

<ParamField path="modelingUnit" type="'cjkchar' | 'bpe' | 'cjkchar+bpe'">
  Modeling unit for hotwords tokenization.
</ParamField>

<ParamField path="bpeVocab" type="string">
  Path to BPE vocabulary file (.vocab). Required when modelingUnit is 'bpe' or 'cjkchar+bpe'.
</ParamField>

<ParamField path="numThreads" type="number" default="1">
  Number of threads for inference.
</ParamField>

<ParamField path="provider" type="string">
  Execution provider (e.g. "cpu").
</ParamField>

<ParamField path="ruleFsts" type="string">
  Path to rule FSTs for inverse text normalization.
</ParamField>

<ParamField path="ruleFars" type="string">
  Path to rule FARs for inverse text normalization.
</ParamField>

<ParamField path="dither" type="number" default="0">
  Dither for feature extraction.
</ParamField>

<ParamField path="modelOptions" type="SttModelOptions">
  Model-specific options. See [SttModelOptions](#sttmodeloptions).
</ParamField>

### SttRecognitionResult

Full recognition result from offline STT.

```typescript
interface SttRecognitionResult {
  text: string;
  tokens: string[];
  timestamps: number[];
  lang: string;
  emotion: string;
  event: string;
  durations: number[];
}
```

<ResponseField name="text" type="string">
  Transcribed text.
</ResponseField>

<ResponseField name="tokens" type="string[]">
  Token strings (word pieces or characters).
</ResponseField>

<ResponseField name="timestamps" type="number[]">
  Timestamps per token (model-dependent).
</ResponseField>

<ResponseField name="lang" type="string">
  Detected or specified language (model-dependent).
</ResponseField>

<ResponseField name="emotion" type="string">
  Emotion label (e.g. SenseVoice models).
</ResponseField>

<ResponseField name="event" type="string">
  Event label (model-dependent).
</ResponseField>

<ResponseField name="durations" type="number[]">
  Durations (valid for TDT models).
</ResponseField>

### Hotwords Support

Only transducer and nemo_transducer models support hotwords.

```typescript
import { STT_HOTWORDS_MODEL_TYPES, sttSupportsHotwords } from 'react-native-sherpa-onnx';

console.log(STT_HOTWORDS_MODEL_TYPES);
// ['transducer', 'nemo_transducer']

const supportsHotwords = sttSupportsHotwords('transducer');
console.log(supportsHotwords); // true

const doesNotSupport = sttSupportsHotwords('whisper');
console.log(doesNotSupport); // false
```

### EndpointConfig

Endpoint detection configuration with three rules.

```typescript
interface EndpointConfig {
  rule1?: EndpointRule;
  rule2?: EndpointRule;
  rule3?: EndpointRule;
}
```

<ParamField path="rule1" type="EndpointRule">
  First endpoint rule (e.g. 2.4s trailing silence, no speech required).
</ParamField>

<ParamField path="rule2" type="EndpointRule">
  Second endpoint rule (e.g. 1.4s trailing silence, speech required).
</ParamField>

<ParamField path="rule3" type="EndpointRule">
  Third endpoint rule (e.g. max utterance length 20s).
</ParamField>

### EndpointRule

Single endpoint detection rule.

```typescript
interface EndpointRule {
  mustContainNonSilence: boolean;
  minTrailingSilence: number;
  minUtteranceLength: number;
}
```

<ParamField path="mustContainNonSilence" type="boolean" required>
  If true, rule only matches when the segment contains non-silence (speech).
</ParamField>

<ParamField path="minTrailingSilence" type="number" required>
  Minimum trailing silence duration in seconds to trigger endpoint.
</ParamField>

<ParamField path="minUtteranceLength" type="number" required>
  Minimum utterance length in seconds (acts as max length cap for rule3).
</ParamField>

**Example:**

```typescript
const endpointConfig: EndpointConfig = {
  // Rule 1: 2.4 seconds of silence (regardless of speech)
  rule1: {
    mustContainNonSilence: false,
    minTrailingSilence: 2.4,
    minUtteranceLength: 0,
  },
  // Rule 2: 1.4 seconds of silence after speech
  rule2: {
    mustContainNonSilence: true,
    minTrailingSilence: 1.4,
    minUtteranceLength: 0,
  },
  // Rule 3: Maximum utterance length of 20 seconds
  rule3: {
    mustContainNonSilence: false,
    minTrailingSilence: 0,
    minUtteranceLength: 20.0,
  },
};
```

## Engine Interfaces

### SttEngine

Offline STT engine interface (returned by `createSTT()`).

```typescript
interface SttEngine {
  readonly instanceId: string;
  transcribeFile(filePath: string): Promise<SttRecognitionResult>;
  transcribeSamples(
    samples: number[],
    sampleRate: number
  ): Promise<SttRecognitionResult>;
  setConfig(options: SttRuntimeConfig): Promise<void>;
  destroy(): Promise<void>;
}
```

See [createSTT](/api/stt/create-stt) for detailed documentation.

### StreamingSttEngine

Streaming STT engine interface (returned by `createStreamingSTT()`).

```typescript
interface StreamingSttEngine {
  readonly instanceId: string;
  createStream(hotwords?: string): Promise<SttStream>;
  destroy(): Promise<void>;
}
```

See [Streaming STT](/api/stt/streaming-stt) for detailed documentation.

### SttStream

Streaming recognition stream interface.

```typescript
interface SttStream {
  readonly streamId: string;
  acceptWaveform(samples: number[], sampleRate: number): Promise<void>;
  inputFinished(): Promise<void>;
  decode(): Promise<void>;
  isReady(): Promise<boolean>;
  getResult(): Promise<StreamingSttResult>;
  isEndpoint(): Promise<boolean>;
  reset(): Promise<void>;
  release(): Promise<void>;
  processAudioChunk(
    samples: number[] | Float32Array,
    sampleRate: number
  ): Promise<{ result: StreamingSttResult; isEndpoint: boolean }>;
}
```

See [Streaming STT](/api/stt/streaming-stt) for detailed documentation.
